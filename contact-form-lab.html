フォーム送信 → 保存 + 運営通知。その関数、Step Functions の直接統合だけで書けます。

// あなたが書いた Lambda — handler.py（抜粋・約20行）

import json, uuid, boto3
ddb = boto3.client('dynamodb'); ses = boto3.client('ses')

def handler(event, context):
    form = json.loads(event['body'])                      # ①
    if not form.get('name') or '@' not in form['email']:  # ②
        return {'statusCode': 403, 'body': '{"ok":false}'}
    ddb.put_item(TableName='inquiries', Item={             # ③
        'id':      {'S': str(uuid.uuid4())},
        'name':    {'S': form['name']},
        'email':   {'S': form['email']},
        'message': {'S': form['message']},
    })
    ses.send_email(Source=FROM,                           # ④
        Destination={'ToAddresses': [ADMIN]},
        Message={'Subject': {'Data': '新しいお問い合わせ: ' + form['name']},
                 'Body':    {'Text':    {'Data': form['message']}}})
    return {'statusCode': 200, 'body': '{"ok":true}'}      # ⑤

// 置き換えた ASL — state machine（抜粋・JSONata）

{
  "QueryLanguage": "JSONata",                       // ① body parse 不要（API GW が Input に渡す）
  "StartAt": "Validate",
  "States": {
    "Validate": {                                    // ②
      "Type": "Choice",
      "Choices": [{ "Next": "Persist", "Condition":
        "{% $exists($states.input.name) and $contains($states.input.email,'@') %}" }],
      "Default": "Invalid"
    },
    "Persist": {                                     // ③④  Parallel で同時実行
      "Type": "Parallel",
      "Branches": [
        { "StartAt": "PutItem", "States": { "PutItem": {        // ③
          "Type": "Task", "Resource": "...:dynamodb:putItem",
          "Arguments": { "TableName": "...", "Item": {
            "id":   { "S": "{% $uuid() %}" },
            "name": { "S": "{% $states.input.name %}" } } }, "End": true }}},
        { "StartAt": "SendEmail", "States": { "SendEmail": {    // ④
          "Type": "Task", "Resource": "...:aws-sdk:ses:sendEmail",
          "Arguments": { "Source": "...", "Destination": { "ToAddresses": ["..."] },
            "Message": { "Subject": { "Data":
              "{% '新しいお問い合わせ: ' & $states.input.name %}" } } }, "End": true }}}
      ],
      "Next": "Respond"
    },
    "Respond": { "Type": "Pass", "Output": { "ok": true }, "End": true }   // ⑤
  }
}
